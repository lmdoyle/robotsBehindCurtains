AFRAME.registerComponent('decrease-click', {
  init: function () {
    // Initialize variables
    this.isButtonPressed = {};
    this.intervalId = {};

    // Prevent mousedown event from propagating to the cursor
    this.el.sceneEl.querySelector('#cursor').addEventListener('mousedown', function (event) {
      event.stopPropagation();
    });

    // Add mousedown event listener to the button
    this.el.addEventListener('mousedown', (event) => {
      // Check if the button is the target of the event
      if (event.target.classList.contains("clickable")) {
        const buttonId = event.target.id;
        
        // Set flag to indicate button press
        this.isButtonPressed[buttonId] = true;

        // Start increasing the value repeatedly while the button is pressed
        this.intervalId[buttonId] = setInterval(() => {
          this.increaseValue(buttonId);
        }, 5); // Adjust interval as needed
      }
    });

    // Add mouseup event listener to the button
    this.el.addEventListener('mouseup', (event) => {
      // Check if the button is the target of the event
      if (event.target.classList.contains("clickable")) {
        const buttonId = event.target.id;
        
        // Reset flag to indicate button release
        this.isButtonPressed[buttonId] = false;

        // Stop increasing the value
        clearInterval(this.intervalId[buttonId]);
      }
    });
  },

    // Initialize variables
    this.isButtonPressed = {};
    this.intervalId = {};

    // Prevent mousedown event from propagating to the cursor
    this.el.sceneEl.querySelector('#cursor').addEventListener('mousedown', function (event) {
      event.stopPropagation();
    });

    // Add mousedown event listener to the button
    this.el.addEventListener('mousedown', (event) => {
      // Check if the button is the target of the event
      if (event.target.classList.contains("clickable")) {
        const buttonId = event.target.id;
        
        // Set flag to indicate button press
        this.isButtonPressed[buttonId] = true;

        // Start increasing the value repeatedly while the button is pressed
        this.intervalId[buttonId] = setInterval(() => {
          this.increaseValue(buttonId);
        }, 5); // Adjust interval as needed
      }
    });

    // Add mouseup event listener to the button
    this.el.addEventListener('mouseup', (event) => {
      // Check if the button is the target of the event
      if (event.target.classList.contains("clickable")) {
        const buttonId = event.target.id;
        
        // Reset flag to indicate button release
        this.isButtonPressed[buttonId] = false;

        // Stop increasing the value
        clearInterval(this.intervalId[buttonId]);
